@expense_bp.route("/gastos", methods=["GET", "POST"])
def gastos():
    """
    Muestra y gestiona los gastos de un usuario.

    **GET:**
      - Lista los gastos del usuario seleccionado.
      - Permite cambiar de usuario mediante el parámetro `user_id`.

    **POST:**
      - Crea un nuevo gasto tras validar los datos enviados desde el formulario.

    :query user_id: (opcional) ID del usuario a consultar. Default = 1.
    :type user_id: str | int

    :form producto: Nombre del producto del gasto.
    :type producto: str
    :form monto: Monto asociado al gasto.
    :type monto: float | str
    :form categoria: Categoría del gasto.
    :type categoria: str
    :form es_hormiga: Si está presente, el gasto se marca como "hormiga".
    :type es_hormiga: bool

    :return: Render de la plantilla `gastos.html` con:
        - `user_id` (int): ID del usuario activo.
        - `gastos` (list): Lista de gastos asociados.
        - `categorias` (list): Categorías válidas.
        - `errores` (dict): Diccionario de errores de validación (si los hay).
    :rtype: flask.Response
    """
    user_id = request.values.get("user_id", "1")
    try:
        user_id = int(user_id)
    except (TypeError, ValueError):
        user_id = 1

    errores = {}
    if request.method == "POST":
        producto = request.form.get("producto", "")
        monto = request.form.get("monto", "")
        categoria = request.form.get("categoria", "otros")
        es_hormiga = True if request.form.get("es_hormiga") == "on" else False

        exito, mensajes = ExpenseService.crear_gasto(user_id, producto, monto, categoria, es_hormiga)

        if not exito:
            errores = mensajes
        else:
            errores = {"general": mensajes.get("general", "Gasto creado.")}

    gastos = ExpenseService.listar_gastos(user_id)
    return render_template(
        "gastos.html",
        user_id=user_id,
        gastos=gastos,
        categorias=CATEGORIAS_VALIDAS,
        errores=errores
    )

@expense_bp.route("/gastos/<int:gasto_id>/editar", methods=["GET", "POST"])
def editar_gasto(gasto_id):
    """
    Muestra y actualiza un gasto específico.

    **GET:**
      - Renderiza el formulario de edición de un gasto.

    **POST:**
      - Procesa los cambios enviados y actualiza el gasto en la base de datos.

    :param gasto_id: ID del gasto a editar.
    :type gasto_id: int
    :query user_id: ID del usuario propietario. Default = 1.
    :type user_id: str | int

    :form producto: Nuevo nombre del producto (opcional).
    :type producto: str
    :form monto: Nuevo monto (opcional).
    :type monto: float | str
    :form categoria: Nueva categoría (opcional).
    :type categoria: str
    :form es_hormiga: Si está presente, se marca como gasto hormiga.
    :type es_hormiga: bool | None

    :return: Render de la plantilla `gastos_editar.html` con:
        - `user_id` (int): ID del usuario.
        - `gasto` (Expense): Gasto a editar.
        - `categorias` (list): Categorías válidas.
        - `errores` (dict): Diccionario con errores si los hay.
    :rtype: flask.Response
    """
    user_id = request.values.get("user_id", "1")
    try:
        user_id = int(user_id)
    except (TypeError, ValueError):
        user_id = 1

    gasto = ExpenseService.obtener_gasto(user_id, gasto_id)
    if not gasto:
        return redirect(url_for("expense.gastos") + f"?user_id={user_id}")

    errores = {}
    if request.method == "POST":
        producto = request.form.get("producto")
        monto = request.form.get("monto")
        categoria = request.form.get("categoria")
        es_hormiga = True if request.form.get("es_hormiga") == "on" else False if request.form.get("es_hormiga") else None

        exito, mensajes = ExpenseService.actualizar_gasto(user_id, gasto_id, producto, monto, categoria, es_hormiga)

        if not exito:
            errores = mensajes
        else:
            return redirect(url_for("expense.gastos") + f"?user_id={user_id}")

    return render_template(
        "gastos_editar.html",
        user_id=user_id,
        gasto=gasto,
        categorias=CATEGORIAS_VALIDAS,
        errores=errores
    )
