import re

PATRON_MINUSCULA = r"[a-z]"
PATRON_MAYUSCULA = r"[A-Z]"
PATRON_DIGITO = r"\d"
PATRON_SIMBOLO = r"[@$!%*?&#^()\-_=+\[\]{}|;:,.<>/]"
PATRON_USUARIO = r"^[A-Za-z0-9_]+$"
PATRON_NOMBRE = r"^[A-Za-z]+$"
PATRON_CORREO = re.compile(
    r"^(?!.*\.\.)"
    r"[A-Za-z0-9](?:[A-Za-z0-9._%+-]{0,62}[A-Za-z0-9])?"
    r"@"
    r"(?:[A-Za-z0-9](?:[A-Za-z0-9-]{0,61}[A-Za-z0-9])?\.)+"
    r"[A-Za-z]{2,24}$"
)
LONGITUD_MIN_USUARIO = 3
LONGITUD_MIN_CONTRASEÑA = 8
LONGITUD_MAX = 20

    @staticmethod
    def validar_nombre(nombre: str) -> list[str]:
        """
        Valida el campo de nombre.

        **Reglas:**
          - Debe ser un string no vacío.
          - Debe tener entre 3 y 20 caracteres.
          - Solo puede contener letras (mayúsculas o minúsculas).

        :param nombre: Nombre del usuario.
        :type nombre: str
        :return: Lista de errores encontrados en el nombre.
        :rtype: list[str]
        """
        errores = []
        if not isinstance(nombre, str) or not nombre.strip():
            errores.append("Debes ingresar un nombre.")
        elif len(nombre) < LONGITUD_MIN_USUARIO or len(nombre) > LONGITUD_MAX:
            errores.append("El nombre debe tener entre 3 y 20 caracteres.")
        if not re.match(PATRON_NOMBRE, nombre):
            errores.append("El nombre solo puede contener letras")
        return errores

    @staticmethod
    def validar_usuario(nombre_usuario: str) -> list[str]:
        """
        Valida el nombre de usuario.

        **Reglas:**
          - Debe ser un string no vacío.
          - Longitud entre 3 y 20 caracteres.
          - Puede contener letras, números y guiones bajos (`_`).
          - No puede contener espacios ni símbolos especiales.

        :param nombre_usuario: Nombre de usuario elegido.
        :type nombre_usuario: str
        :return: Lista de errores en el nombre de usuario.
        :rtype: list[str]
        """
        errores = []
        if not isinstance(nombre_usuario, str) or not nombre_usuario.strip():
            errores.append("Debes ingresar un nombre de usuario.")
        elif len(nombre_usuario) < LONGITUD_MIN_USUARIO or len(nombre_usuario) > LONGITUD_MAX:
            errores.append("El usuario debe tener entre 3 y 20 caracteres.")
        if not re.match(PATRON_USUARIO, nombre_usuario):
            errores.append("El usuario solo puede contener letras, números y '_'.")
        return errores

    @staticmethod
    def validar_correo(correo: str) -> list[str]:
        """
        Valida el correo electrónico.

        **Reglas:**
          - No puede estar vacío.
          - Debe cumplir con el formato estándar `usuario@dominio.ext`.
          - El dominio debe ser válido y no contener `..`.
          - La extensión debe tener entre 2 y 24 caracteres alfabéticos.

        :param correo: Correo electrónico.
        :type correo: str
        :return: Lista de errores en el correo.
        :rtype: list[str]
        """
        errores = []
        if not isinstance(correo, str) or not correo.strip():
            errores.append("Debes ingresar un correo")
        elif re.fullmatch(PATRON_CORREO, correo) is None:
            errores.append("Por favor ingrese un correo válido")
        return errores

    @staticmethod
    def validar_contraseña(contraseña: str) -> list[str]:
        """
        Valida la contraseña según reglas de seguridad.

        **Reglas:**
          - No puede estar vacía.
          - Longitud entre 8 y 20 caracteres.
          - Debe contener al menos:
            * Una letra mayúscula.
            * Una letra minúscula.
            * Un número.
            * Un símbolo especial (`@$!%*?&#^()-_=+[]{}|;:,.<>/`).

        :param contraseña: Contraseña elegida por el usuario.
        :type contraseña: str
        :return: Lista de errores en la contraseña.
        :rtype: list[str]
        """
        errores = []
        if not isinstance(contraseña, str) or not contraseña.strip():
            errores.append("Debes ingresar una contraseña.")
        elif len(contraseña) < LONGITUD_MIN_CONTRASEÑA or len(contraseña) > LONGITUD_MAX:
            errores.append("La contraseña debe tener entre 8 y 20 caracteres.")
        if not re.search(PATRON_MAYUSCULA, contraseña):
            errores.append("Debe contener al menos una mayúscula.")
        if not re.search(PATRON_MINUSCULA, contraseña):
            errores.append("Debe contener al menos una minúscula.")
        if not re.search(PATRON_DIGITO, contraseña):
            errores.append("Debe contener al menos un número.")
        if not re.search(PATRON_SIMBOLO, contraseña):
            errores.append("Debe contener al menos un símbolo.")
        return errores

    @staticmethod
    def registrar_usuario(nombre, apellido, nombre_usuario, correo, contraseña, contraseña_de_nuevo):
        """
        Registra un nuevo usuario después de validar todos los campos.

        **Reglas de validación:**
          - Nombre y apellido válidos (solo letras, entre 4 y 20 caracteres).
          - Usuario válido (4-20 caracteres, letras/números/guiones bajos, único).
          - Correo válido y único.
          - Contraseña válida (8-20 caracteres, cumple las reglas de seguridad).
          - La contraseña debe coincidir con la repetida.

        :param nombre: Nombre del usuario.
        :type nombre: str
        :param apellido: Apellido del usuario.
        :type apellido: str
        :param nombre_usuario: Nombre de usuario único.
        :type nombre_usuario: str
        :param correo: Correo electrónico único.
        :type correo: str
        :param contraseña: Contraseña elegida.
        :type contraseña: str
        :param contraseña_de_nuevo: Repetición de la contraseña.
        :type contraseña_de_nuevo: str
        :return: Resultado del registro.
                 `(True, {"general": ["Usuario registrado con éxito."]})` si es exitoso.
                 `(False, {campo: [errores]})` si hay errores.
        :rtype: tuple[bool, dict]
        """
        errores = {
            "nombre": ServicioLogin.validar_nombre(nombre),
            "apellido": ServicioLogin.validar_apellido(apellido),
            "usuario": ServicioLogin.validar_usuario(nombre_usuario),
            "correo": ServicioLogin.validar_correo(correo),
            "contraseña": ServicioLogin.validar_contraseña(contraseña),
            "contraseña_de_nuevo": ServicioLogin.validar_contraseña_de_nuevo(contraseña, contraseña_de_nuevo)
        }

        usuarios = DatosUsuario.cargar_usuarios()
        for u in usuarios:
            if u.nombre_usuario == nombre_usuario:
                errores["usuario"].append("El usuario ya existe.")
                return False, errores
            if u.correo == correo:
                errores["correo"].append("El correo ya existe.")
                return False, errores

        if any(errores.values()):
            return False, errores

        try:
            if contraseña != contraseña_de_nuevo:
                return False, {"contraseña_de_nuevo": ["La contraseña no coincide"]}
            contraseña_hash = generate_password_hash(contraseña)
            nuevo_usuario = Usuario(nombre.strip(), apellido.strip(), nombre_usuario.strip(), correo.strip(), contraseña_hash)
            usuarios.append(nuevo_usuario)
            DatosUsuario.guardar_usuarios(usuarios)
            return True, {"general": ["Usuario registrado con éxito."]}
        except OSError:
            return False, {"general": ["Error al guardar datos. Intenta nuevamente."]}
        except ValueError as e:
            return False, {"general": [str(e)]}

    @staticmethod
    def autenticar_usuario(nombre_usuario, contraseña):
        """
        Autentica un usuario comparando sus credenciales.

        **Reglas:**
          - Ambos campos son obligatorios.
          - Usuario debe existir en la base de datos.
          - Contraseña debe coincidir con el hash almacenado.

        :param nombre_usuario: Nombre de usuario.
        :type nombre_usuario: str
        :param contraseña: Contraseña ingresada.
        :type contraseña: str
        :return: Resultado de la autenticación con mensaje.
        :rtype: tuple[bool, dict]
        """
        if not isinstance(nombre_usuario, str) or not isinstance(contraseña, str):
            return False, {"general": "Credenciales inválidas"}

        if not nombre_usuario.strip() or not contraseña.strip():
            return False, {"general": "Por favor rellene los campos"}

        usuarios = DatosUsuario.cargar_usuarios()
        for usuario in usuarios:
            if usuario.nombre_usuario == nombre_usuario and check_password_hash(usuario.contraseña_hash, contraseña):
                return True, {"exito": "Inicio de sesión exitoso.",
                              "user_id": int(usuario.user_id)}
        return False, {"general": "Usuario o contraseña incorrectos."}
