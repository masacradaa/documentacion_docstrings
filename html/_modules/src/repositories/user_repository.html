class DatosUsuario:
    """
    Repositorio para el manejo de persistencia de usuarios en el archivo ``usuarios.json``.

    Esta clase se encarga de:
    - Cargar los usuarios desde almacenamiento seguro.
    - Guardar usuarios aplicando las protecciones de seguridad mediante :class:`FileSecurity`.
    """



[documentos]
    @staticmethod
    def cargar_usuarios() -> list[Usuario]:
        """
        Carga todos los usuarios almacenados en el archivo JSON.

        Si el archivo no existe o está corrupto, devuelve una lista vacía.

        :return: Lista de instancias de :class:`Usuario` cargadas desde el archivo.
        :rtype: list[Usuario]
        """
        if not os.path.exists(ARCHIVO_USUARIOS):
            return []

        FileSecurity.hacer_solo_lectura(ARCHIVO_USUARIOS)

        try:
            with open(ARCHIVO_USUARIOS, "r", encoding="utf-8") as f:
                datos = json.load(f)
        except (json.JSONDecodeError, OSError):
            return []

        if not isinstance(datos, list):
            return []

        usuarios = []
        for d in datos:
            if (
                isinstance(d, dict)
                and "nombre" in d
                and "apellido" in d
                and "nombre_usuario" in d
                and "correo" in d
                and "contraseña_hash" in d
            ):
                usuarios.append(Usuario.desde_diccionario(d))
            else:
                print("Error, dato corrupto")

        return usuarios

    @staticmethod
    def guardar_usuarios(usuarios: list[Usuario]) -> None:
        """
        Guarda la lista de usuarios en el archivo JSON.

        Aplica las protecciones de seguridad al archivo al finalizar.

        :param usuarios: Lista de instancias de usuarios a persistir.
        :type usuarios: list[Usuario]
        :return: Nada.
        :rtype: None
        """
        FileSecurity.existe_storage(ARCHIVO_USUARIOS)
        FileSecurity.hacer_escritura(ARCHIVO_USUARIOS)

        try:
            with open(ARCHIVO_USUARIOS, "w", encoding="utf-8") as f:
                json.dump(
                    [u.a_diccionario() for u in usuarios],
                    f,
                    indent=2,
                    ensure_ascii=False,
                )
        finally:
            FileSecurity.proteger_archivo(ARCHIVO_USUARIOS)
            FileSecurity.hacer_solo_lectura(ARCHIVO_USUARIOS)

    @staticmethod
    def get_userid(nombre_usuario: str) -> int | None:
        """
        Obtiene el identificador único (user_id) de un usuario a partir de su nombre de usuario.

        :param nombre_usuario: Nombre de usuario que se desea buscar.
        :type nombre_usuario: str
        :return: El ``user_id`` del usuario si existe, o ``None`` si no se encuentra.
        :rtype: int | None
        """
        if not isinstance(nombre_usuario, str) or not nombre_usuario.strip():
            print("Error: nombre_usuario inválido o vacío.")
            return None

        try:
            usuarios = DatosUsuario.cargar_usuarios()
        except Exception as e:
            print(f"Error al cargar usuarios: {str(e)}")
            return None

        for u in usuarios:
            try:
                if u.nombre_usuario == nombre_usuario.strip():
                    return u.user_id
            except AttributeError:
                print("Error: objeto usuario inválido en la lista.")
                continue
        return None

    @staticmethod
    def get_nombre_usuario(user_id: int) -> str | None:
        """
        Obtiene el nombre de usuario (`nombre_usuario`) a partir de su identificador único.

        :param user_id: ID único del usuario que se desea consultar.
        :type user_id: int
        :return: El nombre de usuario si existe, o ``None`` si no se encuentra.
        :rtype: str | None
        """
        try:
            uid = int(user_id)
            if uid <= 0:
                print("Error: user_id debe ser un número positivo.")
                return None
        except (TypeError, ValueError):
            print("Error: user_id inválido, no es convertible a entero.")
            return None

        try:
            usuarios = DatosUsuario.cargar_usuarios()
        except Exception as e:
            print(f"Error al cargar usuarios: {str(e)}")
            return None

        for u in usuarios:
            try:
                if u.user_id == uid:
                    return u.nombre_usuario
            except AttributeError:
                print("Error: objeto usuario inválido en la lista.")
                continue

        return None
