class ExpenseRepository:
    """
    Repositorio para gestionar los gastos de usuarios en un archivo JSON.
    Se encarga de crear, listar, actualizar y eliminar gastos de un usuario,
    garantizando que los datos se guarden de manera segura en la carpeta `storage/`.
    """

    @staticmethod
    def _leer_json() -> dict:
        """
        Lee el archivo JSON de gastos y devuelve su contenido como un diccionario.

        :return: Diccionario con los datos de gastos.
        :rtype: dict
        """
        if not os.path.exists(DATA_PATH):
            return {"usuarios": {}}
        try:
            FileSecurity.hacer_solo_lectura(DATA_PATH)
            with open(DATA_PATH, "r", encoding="utf-8") as f:
                data = json.load(f)
        except (json.JSONDecodeError, OSError):
            return {"usuarios": {}}
        return data if isinstance(data, dict) else {"usuarios": {}}

    @staticmethod
    def _guardar_json(data: dict) -> None:
        """
        Guarda un diccionario en el archivo JSON de gastos,
        asegurando que se proteja después como solo lectura.

        :param data: Diccionario con la estructura de datos a guardar.
        :type data: dict
        """
        FileSecurity.existe_storage(DATA_PATH)
        FileSecurity.hacer_escritura(DATA_PATH)
        try:
            with open(DATA_PATH, "w", encoding="utf-8") as f:
                json.dump(data, f, indent=2, ensure_ascii=False)
        finally:
            FileSecurity.proteger_archivo(DATA_PATH)
            FileSecurity.hacer_solo_lectura(DATA_PATH)

    @classmethod
    def crear_gasto(cls, user_id: int, producto: str, monto: float, categoria: str, es_hormiga: bool = False) -> bool:
        """
        Crea un nuevo gasto asociado a un usuario.

        :param user_id: ID único del usuario propietario del gasto.
        :type user_id: int
        :param producto: Nombre del producto del gasto.
        :type producto: str
        :param monto: Monto del gasto.
        :type monto: float
        :param categoria: Categoría del gasto.
        :type categoria: str
        :param es_hormiga: Indica si el gasto es considerado "hormiga".
        :type es_hormiga: bool, opcional
        :return: True si el gasto fue creado exitosamente.
        :rtype: bool
        """
        data = cls._leer_json()
        uid_key = str(user_id)

        if uid_key not in data["usuarios"]:
            data["usuarios"][uid_key] = []

        lista = data["usuarios"][uid_key]
        nuevo_id = int(datetime.now().timestamp() * 1000)

        nuevo_gasto = Expense(
            id=nuevo_id,
            user_id=user_id,
            producto=producto,
            monto=float(monto),
            categoria=categoria,
            es_hormiga=es_hormiga,
        )

        lista.append(nuevo_gasto.a_diccionario())
        data["usuarios"][uid_key] = lista
        cls._guardar_json(data)
        return True

    @classmethod
    def listar_gastos_usuario(cls, user_id: int) -> list[Expense]:
        """
        Lista todos los gastos de un usuario específico.

        :param user_id: ID único del usuario.
        :type user_id: int
        :return: Lista de objetos Expense.
        :rtype: list[Expense]
        """
        try:
            uid = int(user_id)
        except (TypeError, ValueError):
            return []

        data = cls._leer_json()
        usuarios = data.get("usuarios", {})
        lista = usuarios.get(str(uid), [])
        return [Expense.desde_diccionario(g) for g in lista]

    @classmethod
    def obtener_gasto(cls, user_id: int, gasto_id: int) -> Expense | None:
        """
        Obtiene un gasto específico de un usuario.

        :param user_id: ID único del usuario propietario del gasto.
        :type user_id: int
        :param gasto_id: ID del gasto a buscar.
        :type gasto_id: int
        :return: Objeto Expense si existe, de lo contrario None.
        :rtype: Expense | None
        """
        gastos = cls.listar_gastos_usuario(user_id)
        for g in gastos:
            if g.id == gasto_id:
                return g
        return None

    @classmethod
    def actualizar_gasto(cls, user_id: int, gasto_id: int, nuevos_datos: dict) -> bool:
        """
        Actualiza un gasto existente de un usuario.

        :param user_id: ID único del usuario.
        :type user_id: int
        :param gasto_id: ID del gasto a actualizar.
        :type gasto_id: int
        :param nuevos_datos: Diccionario con los campos a actualizar (producto, monto, categoría, es_hormiga).
        :type nuevos_datos: dict
        :return: True si el gasto fue actualizado exitosamente, False en caso contrario.
        :rtype: bool
        """
        data = cls._leer_json()
        uid_key = str(user_id)

        if "usuarios" not in data or uid_key not in data["usuarios"]:
            return False

        gastos = data["usuarios"][uid_key]
        for g in gastos:
            if g.get("id") == gasto_id:
                if isinstance(nuevos_datos, dict):
                    if "producto" in nuevos_datos:
                        g["producto"] = nuevos_datos["producto"]
                    if "monto" in nuevos_datos:
                        g["monto"] = nuevos_datos["monto"]
                    if "categoria" in nuevos_datos:
                        g["categoria"] = nuevos_datos["categoria"]
                    if "es_hormiga" in nuevos_datos:
                        g["es_hormiga"] = nuevos_datos["es_hormiga"]

                g["id"] = gasto_id
                g["user_id"] = int(user_id)

                data["usuarios"][uid_key] = gastos
                cls._guardar_json(data)
                return True
        return False

    @classmethod
    def eliminar_gasto(cls, user_id: int, gasto_id: int) -> bool:
        """
        Elimina un gasto de un usuario.

        :param user_id: ID único del usuario.
        :type user_id: int
        :param gasto_id: ID del gasto a eliminar.
        :type gasto_id: int
        :return: True si el gasto fue eliminado, False en caso contrario.
        :rtype: bool
        """
        data = cls._leer_json()
        uid_key = str(user_id)

        if "usuarios" not in data or uid_key not in data["usuarios"]:
            return False

        gastos = data["usuarios"][uid_key]
        nuevos = []
        eliminado = False

        for g in gastos:
            if g.get("id") == gasto_id:
                eliminado = True
            else:
                nuevos.append(g)

        if eliminado:
            data["usuarios"][uid_key] = nuevos
            cls._guardar_json(data)

        return eliminado
