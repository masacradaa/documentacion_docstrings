class FileSecurity:
    """
    Clase utilitaria para gestionar la seguridad de archivos dentro de la carpeta `storage/`.

    Usa comandos de Windows (`icacls` y `attrib`) para aplicar permisos de lectura, escritura
    y protección frente a eliminación, asegurando que los archivos solo puedan ser manipulados
    por cuentas de administrador y el sistema.
    """



[documentos]
    @staticmethod
    def proteger_archivo(path: str) -> None:
        """
        Aplica políticas de seguridad a un archivo:

        - Quita herencia de permisos a la clase padre Usuarios.
        - Concede permisos de lectura y escritura al grupo `Usuarios`.
        - Concede control total a `Administradores` y `SYSTEM`.
        - Quita permisos al grupo `Todos`.
        - Niega explícitamente la eliminación de archivos a `Usuarios`.

        :param path: Ruta absoluta o relativa al archivo a proteger.
        :type path: str
        :return: Nada. Aplica permisos directamente en el sistema de archivos.
        :rtype: None
        """
        if not os.path.exists(path):
            return

        try:
            # Quitar herencia de permisos
            subprocess.run(["icacls", path, "/inheritance:r"],
                           check=True, capture_output=True)

            # Dar permisos explícitos
            subprocess.run(["icacls", path, "/grant:r", "Usuarios:(W,R)"],
                           check=True, capture_output=True)
            subprocess.run(["icacls", path, "/grant:r", "Administradores:F"],
                           check=True, capture_output=True)
            subprocess.run(["icacls", path, "/grant:r", "SYSTEM:F"],
                           check=True, capture_output=True)

            # Quitar permisos a "Todos"
            subprocess.run(["icacls", path, "/remove:g", "Todos"],
                           check=False, capture_output=True)

            # Negar eliminación a Usuarios
            subprocess.run(["icacls", path, "/deny", "Usuarios:(DE)"],
                           check=True, capture_output=True)

        except subprocess.CalledProcessError as e:
            print("Error al aplicar permisos con icacls:", e.stderr.decode())

    @staticmethod
    def hacer_escritura(path: str) -> None:
        """
        Quita el atributo de solo lectura (`+R`) de un archivo para permitir su escritura.

        :param path: Ruta absoluta o relativa al archivo.
        :type path: str
        :return: Nada. Modifica los atributos del archivo en Windows.
        :rtype: None
        """
        if os.path.exists(path):
            try:
                subprocess.run(["attrib", "-R", path], check=True)
            except Exception:
                pass

    @staticmethod
    def existe_storage(path: str) -> None:
        """
        Crea la carpeta `storage/` (o la carpeta padre del archivo) si no existe.

        :param path: Ruta al archivo dentro de la carpeta `storage/`.
        :type path: str
        :return: Nada. Garantiza que la carpeta de almacenamiento exista.
        :rtype: None
        """
        os.makedirs(os.path.dirname(path), exist_ok=True)
