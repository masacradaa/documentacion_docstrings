def crear_app():
    """
    Crea e inicializa la aplicación Flask.

    - Carga la configuración desde :class:`Config`.
    - Registra los blueprints de los distintos controladores.
    - Define un middleware ``after_request`` para evitar el cache en las respuestas.
    - Registra la ruta raíz (``/``) que renderiza ``home.html``.

    :return: Instancia configurada de Flask.
    :rtype: flask.Flask
    """
    app = Flask(__name__)
    app.config.from_object(Config)

    # Registrar blueprints (rutas de controladores)
    registrar_rutas(app)

    @app.after_request
    def evitar_cache(response):
        """
        Middleware que agrega cabeceras para evitar que el navegador cachee las páginas.

        :param response: Objeto de respuesta HTTP.
        :type response: flask.Response
        :return: Respuesta con cabeceras de no-cache.
        :rtype: flask.Response
        """
        response.headers["Cache-Control"] = "no-store, no-cache, must-revalidate, max-age=0"
        response.headers["Pragma"] = "no-cache"
        response.headers["Expires"] = "0"
        return response

    @app.route("/")
    def home():
        """
        Ruta principal de la aplicación.

        :return: Render de la plantilla ``home.html``.
        :rtype: str
        """
        return render_template("home.html")

    return app
