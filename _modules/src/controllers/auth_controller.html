from werkzeug import Response

from src.services.user_login_services import ServicioLogin
import os

auth_bp = Blueprint(
    "auth",
    __name__,
    template_folder=os.path.join(os.path.dirname(__file__), "../templates")
)

@auth_bp.route("/registrarse", methods=["GET", "POST"], endpoint="registrarse")
def registro():
    """
    Vista para registrar un nuevo usuario.

    - En ``GET``: muestra el formulario de registro.
    - En ``POST``: procesa los datos enviados y registra al usuario.

    :return: Renderizado de la plantilla de registro o redirección al login si el registro es exitoso.
    :rtype: str
    """
    errores: dict = {}

    if request.method == "POST":
        nombre: str = request.form.get("nombre")
        apellido: str = request.form.get("apellido")
        nombre_usuario: str = request.form.get("usuario")
        correo: str = request.form.get("correo")
        contraseña: str = request.form.get("contraseña")
        contraseña_de_nuevo: str = request.form.get("contraseña_de_nuevo")

        exito, mensajes = ServicioLogin.registrar_usuario(
            nombre, apellido, nombre_usuario, correo,
            contraseña, contraseña_de_nuevo
        )

        if not exito:
            errores = mensajes
            return render_template(
                "registrarse.html",
                errores=errores,
                datos=request.form
            )

        flash("Usuario registrado correctamente", "exito")
        return redirect(url_for("auth.login"))

    return render_template("registrarse.html", errores=errores)

@auth_bp.route("/login", methods=["GET", "POST"], endpoint="login")
def login():
    """
    Vista para autenticar un usuario.

    - En ``GET``: muestra el formulario de inicio de sesión.
    - En ``POST``: valida credenciales contra el servicio de login y crea sesión si son correctas.

    :return: Renderizado de la plantilla de login o redirección a la vista principal si el login es exitoso.
    :rtype: str
    """
    errores: dict = {}
    error_msg: str | None = None
    exito_msg: str | None = None

    if request.method == "POST":
        nombre_usuario: str = request.form.get("usuario")
        contraseña: str = request.form.get("contraseña")

        exito, mensajes = ServicioLogin.autenticar_usuario(nombre_usuario, contraseña)

        if not exito:
            errores = mensajes.get("general")
            return render_template(
                "login.html",
                errores=errores,
                datos=request.form
            )

        # Guardar sesión
        session["user_id"] = mensajes["user_id"]
        session["nombre_usuario"] = nombre_usuario
        flash("Inicio de sesión exitoso.", "exito")
        return redirect(
            url_for("main.grafico_gastos",
                    user_id=mensajes["user_id"],
                    exito_msg=exito_msg)
        )

    return render_template("login.html",
                           errores=errores,
                           error_msg=error_msg,
                           exito_msg=exito_msg)

@auth_bp.route("/logout")
def logout():
    """
    Cierra la sesión actual del usuario y redirige al inicio.

    :return: Redirección a la vista principal de inicio.
    :rtype: str
    """
    session.clear()
    flash("Sesión cerrada correctamente.", "exito")
    return redirect(url_for("home"))
