class Usuario:
    def __init__(self, nombre: str, apellido: str, nombre_usuario: str,
                 correo: str, contraseña_hash: str, user_id: int | None = None):
        """
        Inicializa un objeto Usuario validando los parámetros de entrada.

        :param nombre: Nombre del usuario.
        :type nombre: str
        :param apellido: Apellido del usuario.
        :type apellido: str
        :param nombre_usuario: Nombre de usuario único.
        :type nombre_usuario: str
        :param correo: Correo electrónico del usuario.
        :type correo: str
        :param contraseña_hash: Contraseña cifrada/hasheada.
        :type contraseña_hash: str
        :param user_id: ID único del usuario (si no se entrega, se genera automáticamente).
        :type user_id: int | None

        :raises ValueError: Si alguno de los parámetros no es un string.

        :ivar user_id: Id del usuario.
        :vartype user_id: int
        :ivar nombre: Nombre del usuario.
        :vartype nombre: str
        :ivar apellido: Apellido del usuario.
        :vartype apellido: str
        :ivar nombre_usuario: Nombre de usuario único.
        :vartype nombre_usuario: str
        :ivar correo: Correo electrónico.
        :vartype correo: str
        :ivar contraseña_hash: Contraseña cifrada/hasheada.
        :vartype contraseña_hash: str
        """
        if not isinstance(nombre, str):
            raise ValueError("El nombre debe ser un string")
        if not isinstance(apellido, str):
            raise ValueError("El apellido debe ser un string")
        if not isinstance(nombre_usuario, str):
            raise ValueError("El nombre de usuario debe ser un string")
        if not isinstance(correo, str):
            raise ValueError("El correo debe ser un string")
        if not isinstance(contraseña_hash, str):
            raise ValueError("La contraseña debe ser un string")

        self.user_id = user_id or int(time.time() * 100)
        self.nombre = nombre.strip()
        self.apellido = apellido.strip()
        self.nombre_usuario = nombre_usuario.strip()
        self.correo = correo.strip()
        self.contraseña_hash = contraseña_hash

    def a_diccionario(self):
        """
        Convierte el objeto Usuario en un diccionario.

        :return: Diccionario con los atributos del usuario.
        :rtype: dict
        """
        return {
            "user_id": self.user_id,
            "nombre": self.nombre,
            "apellido": self.apellido,
            "nombre_usuario": self.nombre_usuario,
            "correo": self.correo,
            "contraseña_hash": self.contraseña_hash
        }

    @staticmethod
    def desde_diccionario(data):
        """
        Crea una instancia de Usuario a partir de un diccionario.

        :param data: Diccionario con las claves requeridas.
        :type data: dict
        :return: Nueva instancia de Usuario.
        :rtype: Usuario
        :raises ValueError: Si falta alguna clave obligatoria en el diccionario.
        """
        try:
            return Usuario(
                user_id=data["user_id"],
                nombre=data["nombre"],
                apellido=data["apellido"],
                nombre_usuario=data["nombre_usuario"],
                correo=data["correo"],
                contraseña_hash=data["contraseña_hash"]
            )
        except KeyError as e:
            raise ValueError(f"Falta el campo obligatorio: {e.args[0]}")
