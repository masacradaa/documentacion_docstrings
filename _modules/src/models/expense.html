class Expense:
    def __init__(self, id: int, user_id: int, producto: str, monto: float, categoria: str, es_hormiga: bool, fecha: str = None):
        """
        Inicializa un objeto Gasto, validando los parámetros de entrada.

        :param id: Identificador único del gasto.
        :type id: int
        :param user_id: Identificador del usuario asociado al gasto (user_id persistente).
        :type user_id: int
        :param producto: Nombre del producto o servicio (no vacío).
        :type producto: str
        :param monto: Monto del gasto (positivo).
        :type monto: int | float
        :param categoria: Categoría del gasto (debe estar en ``CATEGORIAS_VALIDAS``).
        :type categoria: str
        :param es_hormiga: Indica si el gasto corresponde a un gasto hormiga.
        :type es_hormiga: bool
        :param fecha: Fecha del gasto en formato "YYYY-MM-DD". Si no se pasa, se usa la actual.
        :type fecha: str, optional

        :raises ValueError: Si alguno de los parámetros no cumple las validaciones.
        """
        if not isinstance(id, int):
            raise ValueError("El id debe ser entero.")
        if not isinstance(user_id, int) or user_id <= 0:
            raise ValueError("El user_id debe ser un entero positivo.")
        if not isinstance(producto, str) or not producto.strip():
            raise ValueError("El producto debe ser un string no vacío.")
        if not isinstance(monto, (int, float)) or monto <= 0:
            raise ValueError("El monto debe ser un número positivo.")
        if categoria not in CATEGORIAS_VALIDAS:
            raise ValueError(f"Categoría inválida. Debe ser una de: {', '.join(CATEGORIAS_VALIDAS)}")
        if not isinstance(es_hormiga, bool):
            raise ValueError("es_hormiga debe ser booleano.")

        self.id = id
        self.user_id = user_id
        self.producto = producto.strip()
        self.monto = float(monto)
        self.categoria = categoria
        self.es_hormiga = es_hormiga
        self.fecha = fecha or datetime.now().strftime("%Y-%m-%d")

    def a_diccionario(self) -> dict:
        """
        Convierte el objeto Gasto en un diccionario.

        :return: Diccionario con los atributos del gasto.
        :rtype: dict
        """
        return {
            "id": self.id,
            "user_id": self.user_id,
            "producto": self.producto,
            "monto": self.monto,
            "categoria": self.categoria,
            "es_hormiga": self.es_hormiga,
            "fecha": self.fecha
        }

    @classmethod
    def desde_diccionario(cls, data: dict) -> "Expense":
        """
        Crea una instancia de la clase a partir de un diccionario.

        :param data: Diccionario con las claves necesarias.
        :type data: dict
        :return: Nueva instancia de Expense.
        :rtype: Expense
        :raises ValueError: Si falta alguna clave obligatoria.
        """
        try:
            return cls(
                id=data["id"],
                user_id=data["user_id"],
                producto=data["producto"],
                monto=data["monto"],
                categoria=data["categoria"],
                es_hormiga=data["es_hormiga"],
                fecha=data.get("fecha")
            )
        except KeyError as e:
            raise ValueError(f"Falta el campo obligatorio: {e.args[0]}")
