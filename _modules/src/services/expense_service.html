    @staticmethod
    def validar_gasto(user_id, producto, monto) -> dict:
        """
        Valida los campos de un gasto antes de registrarlo.

        **Reglas:**
          - `user_id`:
            * Debe ser un número entero mayor a 0.
          - `producto`:
            * Debe ser un string no vacío.
            * No puede exceder los 50 caracteres.
          - `monto`:
            * Debe ser un número válido.
            * Debe ser mayor a 0.
            * No puede superar los 999 billones.

        :param user_id: ID del usuario propietario del gasto.
        :type user_id: int | str
        :param producto: Nombre del producto asociado al gasto.
        :type producto: str
        :param monto: Monto del gasto.
        :type monto: float | str
        :return: Diccionario con los errores de validación encontrados.
        :rtype: dict
        """
        errores = {}

        try:
            user_id = int(user_id)
        except (TypeError, ValueError):
            errores["user_id"] = "User_id inválido."
        if user_id <= 0:
            errores["user_id"] = "User_id inválido."

        if not isinstance(producto, str) or not producto.strip():
            errores["producto"] = "El producto no puede estar vacío."
        elif len(producto) > LIMITE_CARACTERES_PRODUCTO:
            errores["producto"] = "El nombre del producto excede el límite de 50 caracteres."

        try:
            monto = float(monto)
            if monto <= 0:
                errores["monto"] = "Debe ser mayor a 0."
            elif monto > LIMITE_MONTO:
                errores["monto"] = "El monto no puede superar los 999 billones."
        except (TypeError, ValueError):
            errores["monto"] = "Debe ser un número válido."

        return errores

    @staticmethod
    def crear_gasto(user_id, producto, monto, categoria, es_hormiga=False) -> tuple[bool, dict]:
        """
        Crea un nuevo gasto tras validar los datos.

        **Reglas adicionales:**
          - El `user_id` debe existir en la base de usuarios.
          - La `categoria` debe pertenecer a las categorías válidas.

        :param user_id: ID del usuario propietario del gasto.
        :type user_id: int
        :param producto: Nombre del producto.
        :type producto: str
        :param monto: Monto del gasto.
        :type monto: float | str
        :param categoria: Categoría del gasto.
        :type categoria: str
        :param es_hormiga: Indica si es un gasto hormiga.
        :type es_hormiga: bool
        :return: Resultado del registro.
                 `(True, {"general": "Gasto registrado con éxito."})` si fue exitoso.
                 `(False, {campo: "error"})` si falló.
        :rtype: tuple[bool, dict]
        """
        errores = ExpenseService.validar_gasto(user_id, producto, monto)
        if errores:
            return False, errores

        usuarios = DatosUsuario.cargar_usuarios()
        if not isinstance(usuarios, list) or user_id < 1:
            return False, {"general": "El usuario no existe."}

        try:
            exito = ExpenseRepository.crear_gasto(user_id, producto, monto, categoria, es_hormiga)
            return (True, {"general": "Gasto registrado con éxito."}) if exito else (False, {"general": "No se pudo guardar el gasto."})
        except ValueError as e:
            return False, {"general": f"{str(e)}."}
        except OSError:
            return False, {"general": "No se pudo guardar el gasto."}

    @staticmethod
    def actualizar_gasto(user_id, gasto_id, producto=None, monto=None, categoria=None, es_hormiga=None) -> tuple[bool, dict]:
        """
        Actualiza un gasto existente.

        **Reglas:**
          - `producto`: No vacío, máx. 50 caracteres.
          - `monto`: Número válido, > 0 y ≤ 999 billones.
          - `categoria`: Debe estar en `CATEGORIAS_VALIDAS`.
          - `es_hormiga`: Booleano opcional.

        :param user_id: ID del usuario.
        :type user_id: int | str
        :param gasto_id: ID del gasto.
        :type gasto_id: int | str
        :param producto: Nuevo producto.
        :type producto: str | None
        :param monto: Nuevo monto.
        :type monto: float | str | None
        :param categoria: Nueva categoría.
        :type categoria: str | None
        :param es_hormiga: Nuevo valor de gasto hormiga.
        :type es_hormiga: bool | None
        :return: Resultado de la actualización.
                 `(True, {"general": "Gasto actualizado."})` si fue exitoso.
                 `(False, {campo: "error"})` si falló.
        :rtype: tuple[bool, dict]
        """
        errores = {}
        nuevos = {}

        if producto is not None and producto != "":
            if not isinstance(producto, str) or not producto.strip():
                errores["producto"] = "Producto inválido."
            elif len(producto) > LIMITE_CARACTERES_PRODUCTO:
                errores["producto"] = "El nombre del producto excede el límite de 50 caracteres."
            else:
                nuevos["producto"] = producto.strip()

        if monto is not None and monto != "":
            try:
                monto_val = float(monto)
                if monto_val <= 0:
                    errores["monto"] = "El monto debe ser mayor a 0."
                elif monto_val > LIMITE_MONTO:
                    errores["monto"] = "El monto no puede superar los 999 billones."
                else:
                    nuevos["monto"] = monto_val
            except (TypeError, ValueError):
                errores["monto"] = "Monto inválido."

        if categoria is not None and categoria != "":
            if categoria not in CATEGORIAS_VALIDAS:
                errores["categoria"] = "Categoría inválida."
            else:
                nuevos["categoria"] = categoria

        if es_hormiga is not None:
            nuevos["es_hormiga"] = bool(es_hormiga)

        if errores:
            return False, errores

        try:
            exito = ExpenseRepository.actualizar_gasto(int(user_id), int(gasto_id), nuevos)
            return (True, {"general": "Gasto actualizado."}) if exito else (False, {"general": "Gasto no encontrado."})
        except ValueError as e:
            return False, {"general": str(e)}
        except (TypeError, OSError):
            return False, {"general": "Error al actualizar datos."}
